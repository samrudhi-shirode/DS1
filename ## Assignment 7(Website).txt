<script>

/* Server API URL */
const API = "http://localhost:3000";

/* Function called when user clicks button */
async function calc(op) {

  // Get value of A
  let a = document.getElementById("a").value;

  // Get value of B
  let b = document.getElementById("b").value;

  // Check if inputs are empty
  if (a === "" || b === "") {

    document.getElementById("result").innerText =
    "Enter numbers!";

    return;
  }

  try {

    // Send request to server
    let res = await fetch(

      `${API}/${op}?a=${a}&b=${b}`
    );

    // Convert response into JSON
    let data = await res.json();

    // Display result or error
    document.getElementById("result").innerText =

      data.success

      ? "Result: " + data.result

      : data.error;

  } catch {

    // Display error if server not running
    document.getElementById("result").innerText =
    "Server error!";
  }
}
</script>

//server/server.js
const express = require("express");   // Import Express framework

const cors = require("cors");         // Import CORS middleware

// Create Express application
const app = express();


// Enable CORS
// Allows frontend and backend communication
app.use(cors());


// Home route
app.get("/", (req, res) => {

  res.send("Server running");
});


// Route for arithmetic operations
app.get("/:op", (req, res) => {

  // Get operation from URL
  let { op } = req.params;

  // Get values from query parameters
  let { a, b } = req.query;

  // Convert string values into numbers
  a = Number(a);

  b = Number(b);

  // Variable to store result
  let result;

  // Perform operation
  switch(op) {

    // Addition
    case "add":

      result = a + b;

      break;

    // Subtraction
    case "subtract":

      result = a - b;

      break;

    // Multiplication
    case "multiply":

      result = a * b;

      break;

    // Division
    case "divide":

      // Check divide by zero
      if (b === 0)

        return res.json({

          success: false,

          error: "Divide by 0"
        });

      result = a / b;

      break;

    // Invalid operation
    default:

      return res.json({

        success: false,

        error: "Invalid operation"
      });
  }

  // Send successful response
  res.json({

    success: true,

    result
  });
});


// Start server on port 3000
app.listen(3000, () =>

  console.log("http://localhost:3000")
);
